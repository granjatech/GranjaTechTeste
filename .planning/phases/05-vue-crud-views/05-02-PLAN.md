Rust backend API endpoints for this plan:

Granjas:
- GET /api/granjas -> [{ id, codigo, nome, localizacao, capacidadeTotal, tipoProducao, status, usuarioId, usuarioNome, dataCriacao }]
- POST /api/granjas -> { nome, localizacao, capacidadeTotal, tipoProducao }
- PUT /api/granjas/{id} -> { nome, localizacao, capacidadeTotal, tipoProducao }
- DELETE /api/granjas/{id} -> 204

Lotes:
- GET /api/lotes -> [{ id, codigo, granjaId, granjaNome, dataAlojamento, quantidadeInicial, quantidadeAtual, linhagem, status, racaoTipo, dataEncerramento, viabilidade, iep }]
- GET /api/lotes/{id} -> single lote detail
- POST /api/lotes -> { granjaId, dataAlojamento, quantidadeInicial, linhagem, racaoTipo }
- PUT /api/lotes/{id} -> { granjaId, dataAlojamento, quantidadeInicial, linhagem, racaoTipo, status }
- DELETE /api/lotes/{id} -> 204
- GET /api/lotes/{id}/mortalidades -> [{ id, loteId, data, quantidade, causa }]
- POST /api/lotes/{id}/mortalidades -> { data, quantidade, causa }

Estoque:
- GET /api/estoque -> [{ id, nome, categoria, unidade, quantidadeAtual, quantidadeMinima, precoUnitario, fornecedor, observacoes }]
- POST /api/estoque -> { nome, categoria, unidade, quantidadeAtual, quantidadeMinima, precoUnitario, fornecedor, observacoes }
- PUT /api/estoque/{id} -> { nome, categoria, unidade, quantidadeAtual, quantidadeMinima, precoUnitario, fornecedor, observacoes }
- DELETE /api/estoque/{id} -> 204

Financas:
- GET /api/financas -> [{ id, tipo, categoria, descricao, valor, data, granjaId, granjaNome, usuarioId, usuarioNome, dataCriacao }]
- POST /api/financas -> { tipo, categoria, descricao, valor, data, granjaId }
- PUT /api/financas/{id} -> { tipo, categoria, descricao, valor, data, granjaId }
- DELETE /api/financas/{id} -> 204
