Rust backend API endpoints for this plan:

Consumo:
- GET /api/consumo/racao/{loteId} -> [{ id, loteId, data, quantidade, tipo, observacoes }]
- GET /api/consumo/agua/{loteId} -> [{ id, loteId, data, quantidade, tipo, observacoes }]
- POST /api/consumo/racao -> { loteId, data, quantidade, observacoes }
- POST /api/consumo/agua -> { loteId, data, quantidade, observacoes }
- GET /api/consumo/resumo/{loteId} -> { totalRacao, totalAgua, mediaRacaoDiaria, mediaAguaDiaria }

Pesagem:
- GET /api/pesagem/{loteId} -> [{ id, loteId, data, pesoMedio, quantidadeAmostras, observacoes }]
- POST /api/pesagem -> { loteId, data, pesoMedio, quantidadeAmostras, observacoes }
- GET /api/pesagem/resumo/{loteId} -> { totalPesagens, pesoMedioAtual, ganhoMedioDiario }

Sensores:
- GET /api/sensores -> [{ id, nome, tipo, localizacao, granjaId, granjaNome, status }]
- POST /api/sensores -> { nome, tipo, localizacao, granjaId }
- DELETE /api/sensores/{id} -> 204
- GET /api/sensores/{id}/leituras -> [{ id, sensorId, valor, dataHora }]
- POST /api/leituras -> { sensorId, valor }

Sanitario:
- GET /api/sanitario/{loteId} -> [{ id, loteId, data, tipo, descricao, responsavel, observacoes }]
- POST /api/sanitario -> { loteId, data, tipo, descricao, responsavel, observacoes }
- GET /api/sanitario/resumo/{loteId} -> { totalEventos, ultimoEvento, vacinasAplicadas, medicamentosAdministrados }
- GET /api/sanitario/cronograma-vacinacao -> [{ vacina, idadeRecomendada, descricao }]

Lotes (for dropdown):
- GET /api/lotes -> [{ id, codigo, granjaNome, status }]
