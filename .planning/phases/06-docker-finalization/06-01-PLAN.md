**Stage 1 — planner:**
```dockerfile
FROM rust:1-slim-bookworm AS planner
RUN cargo install cargo-chef
WORKDIR /app
COPY . .
RUN cargo chef prepare --recipe-path recipe.json
```

**Stage 2 — cacher:**
```dockerfile
FROM rust:1-slim-bookworm AS cacher
RUN cargo install cargo-chef
RUN apt-get update && apt-get install -y pkg-config libssl-dev && rm -rf /var/lib/apt/lists/*
WORKDIR /app
COPY --from=planner /app/recipe.json recipe.json
RUN cargo chef cook --release --recipe-path recipe.json
```

**Stage 3 — builder:**
```dockerfile
FROM rust:1-slim-bookworm AS builder
RUN apt-get update && apt-get install -y pkg-config libssl-dev && rm -rf /var/lib/apt/lists/*
WORKDIR /app
COPY --from=cacher /app/target target
COPY --from=cacher /usr/local/cargo /usr/local/cargo
COPY . .
RUN cargo build --release
```

**Stage 4 — runtime:**
```dockerfile
FROM debian:bookworm-slim AS runtime
RUN apt-get update && apt-get install -y ca-certificates libssl3 curl && rm -rf /var/lib/apt/lists/*
RUN groupadd -r appuser && useradd -r -g appuser appuser
WORKDIR /app
COPY --from=builder /app/target/release/granjatech-api .
USER appuser
EXPOSE 8080
HEALTHCHECK --interval=30s --timeout=3s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:8080/health || exit 1
CMD ["./granjatech-api"]
```

Create `granjatech-frontend/nginx.conf` with these exact blocks:

```nginx
server {
    listen 80;
    server_name _;
    root /usr/share/nginx/html;
    index index.html;

    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/json application/xml+rss;

    # Security headers (per D-08)
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    # API reverse proxy (per D-06)
    location /api {
        proxy_pass http://backend:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Swagger UI reverse proxy (per D-07)
    location /swagger-ui {
        proxy_pass http://backend:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # OpenAPI spec reverse proxy (per D-07)
    location /api-docs {
        proxy_pass http://backend:8080;
        proxy_set_header Host $host;
    }

    # SPA routing — Vue Router (per D-08)
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Cache static assets (per D-08)
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # No cache for index.html
    location = /index.html {
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate, max-age=0";
    }

    # Health check endpoint (per DOCK-04)
    location /health {
        access_log off;
        return 200 "healthy\n";
        add_header Content-Type text/plain;
    }
}
```

Create `granjatech-frontend/Dockerfile` with two stages per D-04, D-05:

```dockerfile
# Stage 1: Build
FROM node:20-alpine AS build
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm ci --silent
COPY . .
ENV VITE_API_URL=/api
RUN npm run build

# Stage 2: Production with Nginx
FROM nginx:alpine AS production
COPY --from=build /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
    CMD wget --quiet --tries=1 --spider http://localhost:80/health || exit 1
CMD ["nginx", "-g", "daemon off;"]
```

CRITICAL: The build stage must set `ENV VITE_API_URL=/api` BEFORE `RUN npm run build` because Vite embeds env vars at build time. The output directory is `dist/` (Vite), NOT `build/` (CRA).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> nginx | External HTTP traffic enters via port 80 |
| nginx -> backend | Reverse proxy forwards /api requests to internal backend container |
| backend -> postgres | Backend queries database via internal Docker network |
| Docker build context | Files copied into image layers during build |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Information Disclosure | Docker build context | mitigate | .dockerignore excludes .env, .env.*, target/, node_modules/ — secrets never enter image layers |
| T-06-02 | Elevation of Privilege | Runtime container | mitigate | Non-root user (appuser) in Rust runtime stage via USER directive |
| T-06-03 | Spoofing | nginx reverse proxy | mitigate | X-Forwarded-For and X-Real-IP headers set in proxy_pass to preserve client identity |
| T-06-04 | Information Disclosure | nginx security headers | mitigate | X-Frame-Options SAMEORIGIN, X-Content-Type-Options nosniff, X-XSS-Protection block — per D-08 |
| T-06-05 | Tampering | index.html cache | mitigate | no-store, no-cache on index.html ensures users get latest version; static assets use content-hash filenames with 1y cache |
</threat_model>
