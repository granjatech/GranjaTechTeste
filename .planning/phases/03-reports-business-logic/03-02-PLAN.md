```rust
fn validate_date_range(inicio: &DateTime<Utc>, fim: &DateTime<Utc>) -> Result<(), AppError> {
    if inicio > fim {
        return Err(AppError::BadRequest("A data de inicio nao pode ser posterior a data de fim.".into()));
    }
    let diff = (*fim - *inicio).num_days();
    if diff > 365 {
        return Err(AppError::BadRequest("O periodo do relatorio nao pode exceder 365 dias.".into()));
    }
    Ok(())
}
```
